% A \ac{martingale} is a sequence of random variables $X_0, \hdots, X_n$ satisfying $\EE[\vert X_n\vert] < \infty$ and $\EE[X_{n+1} | X_1, \hdots, X_n] = X_n$ for all $n\geq 0$. 
% \end{definition}

% Intuitively, martingales can help model situations that are ``fair''. If a gambler has some fortune $X_n$ at time $n$, their expected fortune at time $n+1$ is still $X_n$.

% \exlabel

% The classic one-dimensional random walk is a martingale. That is, $X_{n+1}\in \{X_n+1, X_n-1\}$, each with probability $1/2$.
